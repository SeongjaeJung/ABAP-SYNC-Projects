MODULE create_alv OUTPUT.
  IF go_con IS INITIAL.
    PERFORM set_alv USING go_con 'CON1' go_alv.
    CREATE OBJECT go_receiver.
    SET HANDLER go_receiver->toolbar_button FOR go_alv.
    SET HANDLER go_receiver->handle_user_command FOR go_alv.
    SET HANDLER go_receiver->button_click FOR go_alv.
    PERFORM display_alv_01.
  ELSE.
    PERFORM display_alv_01.
*    PERFORM refresh_alv_01.
  ENDIF.
  IF go_con2 IS INITIAL.
    PERFORM set_alv USING go_con2 'CON2' go_alv2.
    CREATE OBJECT go_handler.
    SET HANDLER go_handler->on_double_click FOR go_alv2.
  ENDIF.
  PERFORM display_alv_02.
ENDMODULE.

MODULE modify_screen_0300 OUTPUT.
  IF zsassmm0402-rstat = 'Y'.
    LOOP AT SCREEN.
      CASE screen-name.
        WHEN 'ZSASSMM0402-BPID'.
          screen-input = '0'.
        WHEN 'ZSASSMM0402-HTEXT'.
          screen-input = '0'.
      ENDCASE.
      MODIFY SCREEN.
    ENDLOOP.
  ENDIF.
ENDMODULE.

*& Module MODIFY_SCREEN OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE modify_screen OUTPUT.
  IF sy-tcode <> 'ZASSMM040_C_BDC'.
    LOOP AT SCREEN.
      CASE screen-name.
        WHEN 'PLAID'.
          screen-invisible = 1.
          screen-input = 0.
        WHEN 'PLANO'.
          screen-invisible = 1.
          screen-input = 0.
        WHEN 'B_CREATE'.
          screen-invisible = 1.
          screen-input = 0.
      ENDCASE.
      MODIFY SCREEN.
    ENDLOOP.
  ENDIF.
ENDMODULE.
