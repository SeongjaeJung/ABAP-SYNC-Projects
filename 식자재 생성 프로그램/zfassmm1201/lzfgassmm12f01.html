FORM get_food_code  USING    VALUE(pv_fcatg)
                    CHANGING pv_fdid.
  DATA: lv_num TYPE n LENGTH 7.

  CASE pv_fcatg.
    WHEN 'M'.
      PERFORM number_range USING '01'
                           CHANGING lv_num.
    WHEN 'S'.
      PERFORM number_range USING '02'
                       CHANGING lv_num.
    WHEN 'F'.
      PERFORM number_range USING '03'
                       CHANGING lv_num.
    WHEN 'V'.
      PERFORM number_range USING '04'
                       CHANGING lv_num.
    WHEN 'D'.
      PERFORM number_range USING '05'
                       CHANGING lv_num.
    WHEN 'O'.
      PERFORM number_range USING '06'
                       CHANGING lv_num.
    WHEN 'G'.
      PERFORM number_range USING '07'
                       CHANGING lv_num.
  ENDCASE.
  CONCATENATE 'F' lv_num INTO pv_fdid.
ENDFORM.
